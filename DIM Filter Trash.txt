			tag:junk

			/* Sunset items */
			is:sunset
		)
		or
		(
			/* Low rarity equipment */
			(is:weapon or is:armor)
			(is:common or is:uncommon or is:blue)
		)
		or
		(
			/* Low roll armor */
			is:armor
			-is:classitem
			-is:exotic
			-((is:titan) basestat:resilience+recovery:>=20 basestat:intellect:>=20)
			-((is:titan) basestat:resilience+recovery:>=20 basestat:strength:>=20)
			-((is:titan) basestat:resilience+recovery:>=20 basestat:discipline:>=20)
			-((is:warlock) basestat:resilience+recovery:>=20 basestat:discipline:>=20)
			-((is:warlock) basestat:resilience+recovery:>=20 basestat:intellect:>=20)

			-(is:hunter basestat:mobility+recovery:>=20 basestat:intellect:>=20)
			-(is:hunter basestat:mobility+recovery:>=20 basestat:strength:>=20)
		)
		or
		(
			/* Class items that are not modified or masterwork */
			is:classitem
			-is:modded
			-is:masterwork
		)
		or
		(
			/* Duplicate exotic weapons */
			is:weapon
			is:exotic
			is:dupe

		)

			/* Non-crafted weapons that could be crafted */
			is:patternunlocked

		)
		or

			/* Raghav's addition - guns not on wishlist */
			is:weapon -is:wishlistunknown -is:locked  -is:maxpower -is:wishlist 
		)
	)
